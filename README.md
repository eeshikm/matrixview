# matrixview  
## 安装方式  
pip install matrixview  
## 使用方法  
import matrixview as mv  
import numpy as np  
a = np.random.randn(10, 12)  
mv.view(a, 'a')  # 第一个a是要显示的数组，第二个‘a’是数组名
